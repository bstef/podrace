# Preparing ruby API

  Install latest version of ruby: 
  https://github.com/asdf-vm/asdf-ruby 
  cd /pdapi
  run: 
  bundle install
  rails s

# Preparing client nextjs

  Install nodejs  
  cd /pdclient
  run: 
  npm install
  npm run dev
